local TEXTURE_ID = "rbxassetid://100091122003498"

for _, obj in ipairs(workspace:GetDescendants()) do
	if obj:IsA("BasePart") then

		-- Xóa decal cũ
		for _, d in ipairs(obj:GetChildren()) do
			if d:IsA("Decal") then
				d:Destroy()
			end
		end

		-- Thêm decal 6 mặt
		for _, face in ipairs(Enum.NormalId:GetEnumItems()) do
			local decal = Instance.new("Decal")
			decal.Texture = TEXTURE_ID
			decal.Face = face
			decal.Parent = obj
		end

		-- MeshPart thì đổi TextureID luôn nếu có
		if obj:IsA("MeshPart") then
			pcall(function()
				obj.TextureID = TEXTURE_ID
			end)
		end
	end
end
